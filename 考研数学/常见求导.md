(\log_a x)' &= \frac{1}{x \ln a} \quad (a > 0, a \ne 1) \\
(\ln |x|)' &= \frac{1}{x} \\
(\sin x)' &= \cos x \\
(\cos x)' &= -\sin x \\
(\arcsin x)' &= \frac{1}{\sqrt{1 - x^2}} \\
(\arccos x)' &= -\frac{1}{\sqrt{1 - x^2}} \\
(\tan x)' &= \sec^2 x \\
(\cot x)' &= -\csc^2 x \\
(\arctan x)' &= \frac{1}{1 + x^2} \\
(\operatorname{arccot} x)' &= -\frac{1}{1 + x^2} \\
(\sec x)' &= \sec x \tan x \\
(\csc x)' &= -\csc x \cot x \\
\left[\ln\left(x + \sqrt{x^2 + 1}\right)\right]' &= \frac{1}{\sqrt{x^2 + 1}} \\
\left[\ln\left(x + \sqrt{x^2 - 1}\right)\right]' &= \frac{1}{\sqrt{x^2 - 1}}
\end{align*}
